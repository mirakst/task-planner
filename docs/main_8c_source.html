215 void Find_Start_Hour (User user, task *p_task, int assigned_hours[HOURS_PER_DAY], double data[], int day, int do_year) {
216  int i, j,
217  duration,
218  end_hr = 0,
219  best_hr_start = 0,
220  best_hr_end = 0,
221  should_skip_hr = 0;
222  double cur_value = 0.0,
223  value_avg = 0.0,
224  value_avg_max = 0.0,
225  value_avg_min = __DBL_MAX__,
226  value_min = 0.0,
227  value_max = 0.0;
228 
229  duration = (int) ceil(p_task->duration);
230 
231  for (i = 0; i < HOURS_PER_DAY; i++) {
232  /* Calculate end hour for the task and reset variables for each hour */
233  if (day == 365)
234  end_hr = Wrap_Hour(i + duration);
235  else
236  end_hr = i + duration;
237 
238  should_skip_hr = 0;
239  cur_value = 0.0;
240 
241  should_skip_hr = Should_Skip_Hour(user, *p_task, assigned_hours, day, i, end_hr);
242  /* If the user is available in this hour, calculate the avg price values and attempt to assign them */
243  if(!should_skip_hr) {
244  for (j = 0; j < duration; j++) {
245  /* Handle 'overflowing' hours
246  Eks. duration = 3, p_task->duration = 2.5, s√• skal den sidste times pris ganges med den resterende l√¶ngde af tasken (0.5 timer) */
247  if (p_task->duration > j && p_task->duration < (j + 1)) {
248  cur_value += p_task->power * (p_task->duration - j) * data[i + j];
249  }
250  else
251  cur_value += p_task->power * data[i + j];
252  }
253 
254  value_avg = cur_value / duration;
255 
256  if (value_avg > value_avg_max) {
257  value_avg_max = value_avg;
258  value_max = cur_value;
259  }
260  if (value_avg < value_avg_min) {
261  value_avg_min = value_avg;
262  value_min = cur_value;
263  best_hr_start = i;
264  best_hr_end = Wrap_Hour(best_hr_start + j);
265  }
266  }
267  }
268 
269  /* If value_avg_min is unchanged, assume that no start hour was found */
270  if (value_avg_min == __DBL_MAX__ && !do_year) {
271  printf("Could not find a suitable start hour for task: %s\n", p_task->name);
272  return;
273  }
274  Assign_Task(p_task, best_hr_start, best_hr_end, value_min, value_max, assigned_hours, do_year);
275 }

285 int Should_Skip_Hour (User user, task task, int assigned_hours[HOURS_PER_DAY], int day, int start_hr, int end_hr) {
286  int i;
287 
288  /* Skip to next start hour if the user does not ignore availability and is unavailable */
289  if (task.type == passive && !user.ignore_availability && !user.available_hours[start_hr])
290  return 1;
291 
292  /* Prevent active tasks from ending outside the user's available hours */
293  if (task.type == active) {
294  for (i = start_hr; i != end_hr; i++) {
295  if (day == 365)
296  i = Wrap_Hour(i);
297  if ((!user.ignore_availability && !user.available_hours[Wrap_Hour(i)]) || assigned_hours[Wrap_Hour(i)])
298  return 1;
299  }
300  }
301  return 0;
